#pragma once
#include <string>
#include <vector>

namespace config {

struct FileData {
    std::string path;
    int pages = 0;
    int fromRange;
    int toRange;
    bool loaded = false;
};

struct ConfigData {
    std::string printer     = "Microsoft Print to PDF";
    int         copies      = 1;

    std::string printMode   = "Simplex";
    std::string paper       = "A4";
    std::string scale       = "Fit to page (keep aspect ratio)";
    std::string orientation = "Auto";
    std::string collate     = "Collated (1,2,3 | 1,2,3 | 1,2,3)";

    std::vector<FileData> files;
};

}

    HWND m_btnContinue = nullptr;
    HWND m_btnCancel = nullptr;
    HWND m_lblNotification = nullptr;

    HFONT m_font = nullptr;
    HFONT m_fontSmall = nullptr;

    int m_resTarget = 0;
    int m_resCurrent = 0;

    int m_printTarget = 0;
    int m_printCurrent = 0;

    std::wstring m_resLabel = L"Rendering resources...";
    std::wstring m_printLabel = L"Printing...";

    COLORREF m_resColor = RGB(255,255,0);
    COLORREF m_printColor = RGB(0,255,0);

    bool m_allowContinue = true;

    std::function<void()> m_cbContinue;
    std::function<void()> m_cbCancel;

private:
