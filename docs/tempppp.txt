# IDEA:
- khi được gọi Init, Printer sẽ:
+ lưu tham số lại
+ không làm gì thêm

- khi được gọi Print(), Printer sẽ:
+ lấy file bmp từ path
+ xử lí layout (scale + đổi hết sang Portrait):
++ tạo 1 bmp trống với tỉ lệ paper + dpi printer + Portrait (nhớ sử lí để né unprintable)
++ Xét orientation
+++ if (orientation == "Portrait")
++++ skip
+++ if (orientation == "Landscape")
++++ xoay bmp gốc 90 độ
+++ if (orientation == "Auto")
++++ nếu w bmp gốc > h bmp gốc -> xoay 90 độ, không thì skip
++ scale lại bmp gốc vào bmp mới (nhớ sử lí để né unprintable):
+++ scale = "No Scale" -> chỉ đưa vào canvas, không scale
+++ scale = "Fit to page (keep aspect ratio)" -> scale nhưng giữ ratio
+++ scale = "Fill page (ignore aspect ratio)" -> fill canvas, không quan tâm ratio
+ đẩy máy in với paper = cfg.paper và orientation = "Portrait" (ép printer in giống 1:1 nhất có thể):
++ tạo Job:
+++ đẩy ảnh vào (đẩy 2 cái nếu PrintDuplex và 1 nếu PrintSimplex)
+++ chờ flag từ printer hoặc spooler (chọn cách cân bằng giữa chính xác, đơn giản):
++++ nếu thành công -> return 0 error = ""
++++ nếu fail -> return 1 v error 

- ShutDown():
+ cleanup nếu cần

- *note: 
+ ShutDown() chỉ được dùng từ bên ngoài, không tự gọi ShutDown từ trong Printer
+ mỗi Print() là 1 job
+ workflow sẽ là:
++ Init()
++ Print()
++ Print()
++ Print()
++ Print()
++ ...
++ ShutDown()
++ -> không Init / ShutDown mỗi lần Print

std::vector<std::string> GetSupportedPapers(const std::string& printerName)
{
    std::vector<std::string> result;

    std::wstring wPrinter = StringToWString(printerName);

    int count = DeviceCapabilitiesW(
        wPrinter.c_str(),
        nullptr,
        DC_PAPERS,
        nullptr,
        nullptr
    );

    if (count <= 0)
        return result;

    std::vector<WORD> paperIds(count);
    std::vector<wchar_t> paperNames(count * 64); // mỗi name tối đa 64 wchar

    DeviceCapabilitiesW(
        wPrinter.c_str(),
        nullptr,
        DC_PAPERS,
        reinterpret_cast<LPWSTR>(paperIds.data()),
        nullptr
    );

    DeviceCapabilitiesW(
        wPrinter.c_str(),
        nullptr,
        DC_PAPERNAMES,
        paperNames.data(),
        nullptr
    );

    for (int i = 0; i < count; ++i) {
        std::wstring name(&paperNames[i * 64]);
        result.push_back(MapPaperName(paperIds[i], name));
    }

    return result;
}
