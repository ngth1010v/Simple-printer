# FORM
[Basic]
priner=Canon LBP 3018
+ có thể rỗng

[Advance]
printMode=Manual Duplex (Flip On Short Edge)
+ là dạng select, chứa 1 trong các giá trị sau đây (không gổm quote):
++ "Simplex"
++ "Duplex"
++ "Manual Duplex (Flip On Long Edge)"
++ "Manual Duplex (Flip On Short Edge)"

paper=A4
+ khi có printer sẽ update các size theo support của printer 

layout=1 Page Per Sheet
+ là dạng select, chứa 1 trong các giá trị sau đây (không gổm quote):
++ "1 Page Per Sheet"
++ "2 Page Per Sheet"
++ "4 Page Per Sheet"
++ "6 Page Per Sheet"
++ "8 Page Per Sheet"

pagesToPrint=Print All Pages
+ là dạng select, chứa 1 trong các giá trị sau đây (không gổm quote):
++ "Print All Pages"
++ "Only Print Odd Pages"
++ "Only Print Even Pages"

orientation=Portrait
+ là dạng select, chứa 1 trong các giá trị sau đây (không gổm quote):
++ "Auto"
++ "Protrait"
++ "Landscape"

collate=Collated (1,2,3 | 1,2,3 | 1,2,3)
+ là dạng select, chứa 1 trong các giá trị sau đây (không gổm quote):
++ "Collated (1,2,3 | 1,2,3 | 1,2,3)"
++ "Uncollated (1,1,1 | 2,2,2 | 3,3,3)"

[Margin]
top=1
bottom=1
left=1
right=1
+ tất cả tính theo inch
